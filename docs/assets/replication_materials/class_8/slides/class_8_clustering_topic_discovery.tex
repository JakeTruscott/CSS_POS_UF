\section{Discriminating Words}\label{discriminating-words}

\begin{frame}{Discriminating Words}
\phantomsection\label{discriminating-words-1}
\begin{itemize}
\item Not \textit{discriminatory} words... \par \vspace{2.5mm} \pause
\item \textbf{Discrimination Words}: The frequency and distribution of words carry meaningful information about latent categories – i.e., words that appear more frequently in some categories than others are informative signals about that category  \par \vspace{2.5mm}
\item Ex: \textit{Healthcare Provider} vs. \textit{Abortionist} -- Terms used in the same debate but relay competing sentiments/connotations/frames
\end{itemize}
\end{frame}

\end{itemize}

\begin{frame}{Mutual Information}
\phantomsection\label{mutual-information}
\begin{itemize}
\item \textbf{Mutual Information}: The amount of information one variable provides about another.  \pause \par \vspace{2.5mm}
\item MI = 0 $\sim$ Information between two variables (ex: a label and a word) are completely independent (bad discriminator) \pause \par \vspace{2.5mm}
\item MI = 1 $\sim$ Knowing about one variable completely informs us of the other (good discriminator). 
\end{itemize}
\end{frame}

\begin{frame}{Calculating Mutual Information}
\phantomsection\label{calculating-mutual-information}
\begin{itemize}
\item $k$ = Categories
\item $j$ = Word
\item $\pi_k$ = Proportion of Documents for $k$
\item $\pi_{-k}$ = Probability that Doc Doesn't Belong in $k$
\end{itemize}
\end{frame}

\begin{frame}{Calculating Mutual Information (Cont.)}
\phantomsection\label{calculating-mutual-information-cont.}
\begin{center}
 $H(k) = -\pi_k\log_2\pi_k-\pi_{-k}\log_2\pi_{-k,} $ \par \vspace{2.5mm} \textbf{Unconditional Certainty}:   \footnotesize{\textit{How likely we are to assume any random document belongs to a particular category, without even seeing a word}} \par \vspace{5mm} \pause \normalsize
 $H(k|j) = -\sum_{k,-k}\sum_{j,-j}\pi_{k,j}\log_2\pi_{k,j} $ \\  Conditional Certainty  \par \vspace{5mm} \pause
 $\text{MI}_{kj} = H(k) - H(k|j)$ \\ Mutual Information = Unconditional Certainty - Conditional Certainty
\end{center}
\end{frame}

\begin{frame}{Fightin' Words (Monroe, Coloresi, and Quinn 2008)}
\phantomsection\label{fightin-words-monroe-coloresi-and-quinn-2008}
\begin{itemize}
\item Alternative strategy to identify words that discriminate b/w categories using regularized log-odds ratios: $\alpha_j$ = Dirichlet prior \& $W^*_{k} \sim \text{Multinomial}(n_k,\mu_k)$ 
\item[] $$\hat{\mu}_{jk} = \frac{W^*_{jk} + \alpha_j}{n_k + \sum^j_{j=1}\alpha_j}$$ \par \vspace{2.5mm}
\item MI identifies words that are most informative about category membership overall -- Fightin’ Words estimates which words are disproportionately distinctive between groups by accounting for baseline frequencies and sampling variability.
\item Estimate corresponds to expected value of posterior distribution $p(\mu|W)$ -- $\hat{\mu}_{jk}$ is smoothed probability of word $j$ in category $k$, allowing for small prior (e.g., Dirichlet). 
\item[] $$ \text{Log Odds Ratio}_{kj} = \log(\frac{\mu_{kj}}{1-\mu_{kj}}) - \log(\frac{\mu-kj}{1-\mu_{-kj}}) $$
\end{itemize}
\end{frame}

\begin{frame}{Fightin' Words (Monroe, Coloresi, and Quinn 2008)}
\phantomsection\label{fightin-words-monroe-coloresi-and-quinn-2008-1}
\begin{columns}
\column{0.4\textwidth}
\centering
\begin{tabular}{lccc}
\toprule
Word & Red & Blue & Total \\
\midrule
Apple & 4 & 0 & 4 \\
Tree  & 3 & 1 & 4 \\
Green & 2 & 0 & 2 \\
Juice & 1 & 0 & 1 \\
Blue  & 0 & 3 & 3 \\
Ocean & 0 & 4 & 4 \\
Water & 0 & 2 & 2 \\
\bottomrule
\end{tabular}
\column{0.6\textwidth}
\begin{itemize}
\item Assuming $\alpha_j$ = 0.5 for all -- Let's do \textit{Apple} \par \vspace{2.5mm} \pause
\item $\hat{\mu}(Apple|Red) = \frac{n_{(Apple,Red)} + \alpha_{Apple}}{n_{Red} + \sum_j\alpha_j}$ \par \vspace{2.5mm}
\item $\hat{\mu}(Apple|Red) = \frac{4 + 0.5}{10 + 7 \times 0.5} = 0.33 $ \par \vspace{2.5mm} \pause
\item \textbf{Let's solve for $\hat{\mu}(Apple|Blue)$} \par \vspace{2.5mm} \pause
\item $\hat{\mu}(Apple|Blue) = \frac{0 + 0.05}{9 + 7 \times 0.5} = 0.04 $
\end{itemize}
\vfill
\end{columns}
\end{frame}

\begin{frame}{Fightin' Words (Cont.)}
\phantomsection\label{fightin-words-cont.}
\begin{itemize}
\item Now that we have $\hat{\mu}(\text{Apple}|\text{Red/Blue})$, the log odds ratio that \textbf{Apple} is a predictor of \textbf{Red} is: \par \vspace{2.5mm} \pause
\item $\delta{Red} = \log(\frac{0.33}{1-0.33}) \rightarrow \log(0.492) \approx -0.70 $ \par \vspace{2.5mm}
\item  $\delta{Blue} = \log(\frac{0.04}{1-0.04}) \rightarrow \log(0.041) \approx -3.19$ \par \vspace{2.5mm}
\item $\delta_{kj} = -0.70 - (-3.19) \approx 2.49$ \pause \par \vspace{2.5mm}
\item $exp(2.49) \approx$ 12x -- Meaning that the odds \textbf{Apple} appears in \textbf{Red} documents is approx. 12 times larger than \textbf{Blue} -- because it's regularized we know its not just the result of an imabalance in observations for Red vs. Blue
\end{itemize}
\end{frame}

\begin{frame}{K-Means Steps (Canonical)}
\phantomsection\label{k-means-steps-canonical}
\begin{enumerate}
\item Choose groups $k$ -- ex: 3 groups of documents, topics, etc. \pause
\item Start w/ Random Group Centers ($\mu_k$) -- The algorithm will randomly assign groups to start \pause (temporary guesses just to get the ball rolling...)
\item Assign Each Document to Nearest Center based on proximity (e.g., eucledian distance) \pause
\item Relocate the centers to the middle (average position in data space) of each the items assigned to it \pause
\item Repeat this process until convergence or threshold met -- algorithm will keep reassigning items until things stabilize (local optimum or minimization of global objective function) \pause
\end{enumerate}
\begin{itemize}
\item[] \textbf{In short...} k-means keeps reshuffling items into groups and repositioning the group centers until it finds a stable set of clusters where items in the same group are more similar to each other than to items in other groups.
\end{itemize}
\end{frame}

\phantomsection\label{k-means-text-imdb-data-notes}
\begin{itemize}
\item \textbf{Remember}: Text data is high-dimensional -- not always possible to neatly cluster, especially when models are unsupervised. \par \vspace{2.5mm}
\item Selection of clusters $k$ motivated by \textbf{Theory} and Subsequent \textbf{Validation} (multiple mechanisms to measure improvement technically and otherwise -- e.g., \textit{does clustering make sense?}) \par \vspace{2.5mm}
\item Goal is to locate \textbf{local optimum} -- usually a convergence point where shuffling to minimize sum of squares again yields an improvement less than some determined amount. 
\item This isn't linear algebra -- There can be multiple ``correct'' solutions...
\end{itemize}
\end{frame}

\begin{frame}{Clustering -- Other Notes}
\phantomsection\label{clustering-other-notes}
\begin{itemize}
\item \textbf{Algorithmic v. Probabilistic}: K-means is algorithmic (deterministic assignments via nearest centroid), while other models (e.g., LDA) are probabilistic and allow for documents to belonging in several clusters. \par \vspace{2.5mm} \pause
\item \textbf{Soft v. Hard Partitions}: K-means uses hard partitions to assign to nearest cluster centroid. It will only be assigned to one cluster only. \par \vspace{2.5mm} \pause
\item \textbf{Means v. Mediods}:  In the sense of k-means/mediods, its the difference between measuring (a) the location of the average document in the cluster (sensitive to outliers), or (b) the document that minimizes the distance to all others in the cluster (can theoretically omit behavior we're interested in). \par \vspace{2.5mm} \pause
\item \textbf{Flat v. Hierarchical}:  Hierarchical: Flat clustering methods (like k-means) produce a single-level partition of the data into a fixed number of clusters. Hierarchical clustering produces a nested tree of clusters, either by successively merging individual observations or recursively splitting the dataset.
\end{itemize}
\end{frame}

\section{Topic Models}\label{topic-models}

\begin{frame}{First Exercise (SOTU Addresses)}
\phantomsection\label{first-exercise-sotu-addresses}
\begin{itemize}
\item \textbf{Recall...} First exercise we did re: text analysis was Presidential SOTU addresses \par \vspace{2.5mm} \pause
\item Important Assumptions:  \pause
  \begin{enumerate}
  \item Presidents may touch on various policy areas and themes throughout these speeches. \par \vspace{2.5mm} \pause
  \item Even though they may draw from different perspectives and contexts, we can define a word or assert a general vocabulary to identify those thematic elements -- e.g., I can use terms like \textbf{navy, army, marines}, etc. to reasonably identify periods where the president is discussing military matters.
  \end{enumerate} \par \vspace{2.5mm} \pause
\item \textbf{Problem}: Clustering algorithms (even with soft partitioning) might obscure results here -- speeches (documents) engage with multiple topics, frames, etc -- single-member clustering can be problematic. 
\end{itemize}
\end{frame}

\begin{frame}{Topic Models v. Clustering}
\phantomsection\label{topic-models-v.-clustering}
\begin{itemize}
\item \textbf{Topic Model}: Rather than assign each document to only one cluster, topic models assign each document with proportional membership to all categories (topics). 
\item That is...topic models suppose that each document is a mixture across categories – a mixed membership model \par \vspace{2.5mm}
\item Ex: Re: SOTU Addresses -- topic models may help us understand this through the lens of more discrete items like the welfare state, taxation, foreign trade or other topics within the broader scope of discussion related to economic policy. \par \vspace{2.5mm}
\item We'll focus on two: LDA \& STM
\end{itemize}
\end{frame}

\begin{frame}{Additional Notes re: Topic Models}
\phantomsection\label{additional-notes-re-topic-models}
\begin{itemize}
\item Note: Like clustering, there is not about estimating the \textit{only} right answer \par \vspace{2.5mm} \pause
\item Much of the inferential value is assessed post-hoc -- recovering useful topics to explain variances in the data recovered through its assignment process. \par \vspace{2.5mm} \pause
\item Like clustering we can measure (for instance) improvement in log-likelihood to assess whether variance in $k$ are an improvement. 
\end{itemize}
\end{frame}

\begin{frame}{LDA (Latent Dirichlet Allocation)}
\phantomsection\label{lda-latent-dirichlet-allocation}
\begin{itemize}
\item Canonical topic model \par \vspace{2.5mm}
\item Generative probabilistic model – explicitly posits a DGP for documents where documents are mixtures of latent topics and words are drawn conditional on those topics. \par \vspace{2.5mm}
\item Tries to Recover: 
  \begin{enumerate}
  \item \textbf{Document-Topic Distributions}: Each document is represented a distribution over (perhaps multiple) different topics, each with a certain probability.
  \item \textbf{Topic-Word Distributions}: Each topic is represented as a distribution over words, meaning that each topic is defined by a set of words (again all with an associated probability of appearing in that topic)
  \end{enumerate} 
\end{itemize}
\end{frame}

\begin{frame}{LDA Notation}
\phantomsection\label{lda-notation}
\begin{itemize}
\item Document Index = $i$ \par \vspace{2.5mm}
\item Topic Index = $K$ \par \vspace{2.5mm}
\item Word Position In Document ($i$) = $m$ \par \vspace{2.5mm}
\item Observed Word = $W_{im}$ \par \vspace{2.5mm}
\item Document-Topic Weights: $\pi_{ik}$ where $\sum_{K}\pi_{iK} =1$ (All Sum to 1) 
\end{itemize}
\end{frame}

\begin{frame}{LDA (Cont.)}
\phantomsection\label{lda-cont.}
\begin{itemize}
\item[] \centering $Z_{im} \sim \text{Multinomial}(1,\pi_i) \quad \text{Topic Indicator for Each Word} $ \par \vspace{2.5mm}
\item[] \centering  $W_{im} \sim \text{Multinomial}(1, \mu{Z_{im}}) \quad \text{Observed Word Token For Each Word Position}$ \par \vspace{2.5mm}
\item \raggedright Probability of observing a word token in a document ($W_{im} = j$) conditional on both the document-topic weights ($\pi_{iK}$) and the topic-word distributions ($\mu_k$) is a weighted average of topic-specific word probabilities, where the weights are given by the document's topic proportions ($\pi_{iK}$): \par \vspace{2.5mm}
\item[] \centering  $ p(W_{im} = j \mid \pi_i,\mu) = \sum_K\pi_{iK}\mu_{jK}$
\end{itemize}
\end{frame}

\begin{frame}{STM - Structural Topic Models}
\phantomsection\label{stm---structural-topic-models}
\begin{itemize}
\item STMs are another form of topic models that provide for the inclusion of covariates to help explain topic prevalence ($X^p_i$) and content ($X^c_i$).  \par \vspace{2.5mm} \pause 
\item For each document -- Model first decides how much of each topic is likely to appear (topic prevalence) – this can be driven by features like authorship, publication date, etc. \par \vspace{2.5mm} \pause 
\item For each word in document -- Model picks a topic according to those proportions and generates the word from that topic. \par \vspace{2.5mm}
\item  It’s a way to combine topic discovery with regression-style analysis, giving you richer insights into both what topics are present and how they’re talked about.
\end{itemize}
\end{frame}
